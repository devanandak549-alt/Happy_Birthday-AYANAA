                runClockCountdown();
            });

            function runClockCountdown() {
                let counter = 5;
                
                function cycle() {
                    if (counter > 0) {
                        countdownNumDisplay.innerText = counter;
                        countdownNumDisplay.classList.remove('pop');
                        void countdownNumDisplay.offsetWidth; // Trigger DOM Reflow context layout loop reset
                        countdownNumDisplay.classList.add('pop');
                        counter--;
                        setTimeout(cycle, 1000);
                    } else {
                        // Extinguish candle safely
                        wickNode.classList.add('blown-out');
                        countdownNumDisplay.style.display = 'none';
                        wishInstruction.style.opacity = '0';
                        
                        // Transition Stage viewscreen context forward
                        setTimeout(() => {
                            sceneWish.style.opacity = '0';
                            setTimeout(() => {
                                sceneWish.style.display = 'none';
                                body.classList.remove('prevent-scroll');
                                mainStage.style.opacity = '1';
                                triggerSceneReveal();
                            }, 1000);
                        }, 1200);
                    }
                }
                cycle();
            }

                const colors = ['#FF69B4', '#9370DB', '#B76E79', '#FFFFFF', '#FFB6C1'];
                const particles = [];

                for (let i = 0; i < 110; i++) {
                    particles.push({
                        x: Math.random() * width,
                        y: Math.random() * -height - 20,
                        r: Math.random() * 6 + 4,
                        d: Math.random() * colors.length,
                        vectorX: Math.random() * 4 - 2,
                        vectorY: Math.random() * 5 + 4,
                        tilt: Math.random() * 10 - 5
                    });
                }

                function renderLoop() {
                    ctx.clearRect(0, 0, width, height);
                    let ongoingAnimation = false;

                    particles.forEach((p) => {
                        p.y += p.vectorY;
                        p.x += p.vectorX;
                        p.tilt += 0.1;

                        if (p.y < height) ongoingAnimation = true;

                        ctx.beginPath();
                        ctx.lineWidth = p.r;
                        ctx.strokeStyle = colors[Math.floor(p.d)];
                        ctx.moveTo(p.x + p.tilt + p.r / 2, p.y);
                        ctx.lineTo(p.x + p.tilt, p.y + p.tilt + p.r / 2);
                        ctx.stroke();
                    });

                    if (ongoingAnimation) {
                        requestAnimationFrame(renderLoop);
                    } else {
                        ctx.clearRect(0, 0, width, height);
                    }
                }
                requestAnimationFrame(renderLoop);
            }

            /* --- STEP 3: PHOTO INTERSECTION CONTROLLER --- */
            const photoObserver = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        cinemaPhotoNode.classList.add('colorized');
                    }
                });
            }, { threshold: 0.4 });

            photoObserver.observe(document.getElementById('scene-photo'));

            /* --- STEP 4: TYPEWRITER TEXT LOGIC (FIXED SPACES) --- */
            const letterTextContent = "Dear Ayana,\n\nToday is a reminder that the world became brighter because you were born.\n\nEven though we're miles apart now, some people never truly feel distant.\n\nThe memories we made, the laughter we shared, and the friendship we built continue to stay close no matter where life takes us.\n\nI may not be there beside you today, but I am still celebrating you from afar.\n\nThank you for being such an important part of my life.\n\nI hope this year brings you endless happiness, beautiful moments, and dreams that come true.\n\nHappy Birthday, Ayana ❤️";

            let letterTriggered = false;
            const letterObserver = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting && !letterTriggered) {
                        letterTriggered = true;
                        executeCinemaTypewriter(typewriterOutput, letterTextContent, 45);
                    }
                });
            }, { threshold: 0.2 });

            letterObserver.observe(document.getElementById('scene-letter'));

            function executeCinemaTypewriter(targetElement, fullText, speedDelay) {
                let currentIdx = 0;
                
                function typeNode() {
                    if (currentIdx < fullText.length) {
                        // Use textContent directly to safely retain raw text spaces and newlines natively
                        targetElement.textContent += fullText.charAt(currentIdx);
                        currentIdx++;
                        setTimeout(typeNode, speedDelay);
                    }
                }
                typeNode();
            }

            let giftOpened = false;
            giftClicker.addEventListener('click', () => {
                if (giftOpened) return;
                giftOpened = true;
                
                giftClicker.classList.add('opened');

            function triggerBurstExplosionHearts() {
                for(let i=0; i<15; i++) {
                    const heart = document.createElement('div');
                    heart.className = 'global-floating-heart';
                    heart.innerHTML = Math.random() > 0.5 ? '❤️' : '✨';
                    heart.style.left = '50vw';
                    heart.style.top = '50vh';
                    heart.style.opacity = '1';
                    
                    document.body.appendChild(heart);

                    const angle = Math.random() * Math.PI * 2;
                    const velocity = Math.random() * 120 + 60;
                    const destX = Math.cos(angle) * velocity;
                    const destY = Math.sin(angle) * velocity - 100; // Force upward movement bias

                    heart.animate([
                        { transform: 'translate(-50%, -50%) scale(0.5)', opacity: 1 },
                        { transform: `translate(calc(-50% + ${destX}px), calc(-50% + ${destY}px)) scale(1.3)`, opacity: 0 }
                    ], {
                        duration: 1200,
                        easing: 'cubic-bezier(0.1, 0.8, 0.3, 1)',
                        fill: 'forwards'
                    });

                    setTimeout(() => heart.remove(), 1200);
                }
            }

            /* --- STEP 6: STARRY COSMOS SKY ENGINE --- */
            function initStarfieldSky() {
                const ctx = starfieldCanvas.getContext('2d');
                let w = starfieldCanvas.width = window.innerWidth;
                let h = starfieldCanvas.height = window.innerHeight;

                const starNodes = [];
                for(let i=0; i<65; i++) {
                    starNodes.push({
                        x: Math.random() * w,
                        y: Math.random() * h,
                        size: Math.random() * 1.5 + 0.5,
                        opacity: Math.random(),
                        speed: Math.random() * 0.02 + 0.005
                    });
                }

                function starLoop() {
                    ctx.clearRect(0,0,w,h);
                    starNodes.forEach(s => {
                        s.opacity += s.speed;
                        if(s.opacity > 1 || s.opacity < 0) s.speed = -s.speed;
                        ctx.fillStyle = `rgba(255,255,255,${Math.abs(s.opacity)})`;
                        ctx.beginPath();
                        ctx.arc(s.x, s.y, s.size, 0, Math.PI*2);
                        ctx.fill();
                    });
                    requestAnimationFrame(starLoop);
                }
                requestAnimationFrame(starLoop);
            }
            initStarfieldSky();
